172 void encodeHeaders_to(std::string& rs) const
173 {
174 for (auto& [k, v] : rrd["headers"].items()) {
175 if (v.is_string()) {
176 std::format_to(std::back_inserter(rs), "{}: {}\r\n", k, v.get<std::string>());
177 }
178 else if (v.is_number_unsigned()) {
179 std::format_to(std::back_inserter(rs), "{}: {}\r\n", k, v.get<uint64_t>());
180 }
181 else if (v.is_number_integer()) {
182 std::format_to(std::back_inserter(rs), "{}: {}\r\n", k, v.get<int>());
183 }
184 else {
185 std::format_to(std::back_inserter(rs), "{}: {}\r\n", k, v.dump());
186 }
187 }
188
189 std::format_to(std::back_inserter(rs), "\r\n");
190 }

195 std::string encode() const
196 {
197 std::string rs;
198 std::string body;
199 auto& hs = rrd.at("headers");
200 auto& rl = rrd.at("request");
201
202 // Request Line
203 std::format_to(std::back_inserter(rs),
204 "{} {} {}\r\n",
205 rl["method"].get<std::string>(),
206 rl["uri"].get<std::string>(),
207 rl["version"].get<std::string>());
208
209 // Build the content to ensure we have the content-type
210 if (rrd["content"].is_object()) {
211 body = rrd["content"].dump();
212 }
213 else if (rrd["content"].is_string()) {
214 body = rrd["content"].get<std::string>();
215 }
216
217 // Headers..
218 encodeHeaders_to(rs);
219
220 // Finally the content..
221 if (!body.empty()) {
222 std::format_to(std::back_inserter(rs), "{}", body);
223 }
224
225 return std::move(rs);
226 }

355 {
356 protected:
357 basic_restresponse() { }
358
359 public:
363 basic_restresponse& setContent(const std::string& c)
364 {
365 if (!c.empty()) {
366 if (rrd["headers"].value("Content-Type", "").find("json") != std::string::npos) {
367 try {
368 rrd["content"] = nlohmann::json::parse(c);
369 if (!rrd["headers"].contains("Content-Length")) rrd["headers"]["Content-Length"] = c.length();
370 return *this;
371 }
372 catch (...) {
373 }
374 }
375
376 // We did not decode a json; assign as-is
377 rrd["content"] = c;
378 if (!rrd["headers"].contains("Content-Length")) rrd["headers"]["Content-Length"] = c.length();
379 }
380
381 return *this;
382 }

413 std::string encode() const
414 {
415 std::string rs;
416 std::string body;
417 auto& hs = rrd.at("headers");
418 auto& rl = rrd.at("response");
419
420 // Request Line
421 std::format_to(std::back_inserter(rs),
422 "{} {} {}\r\n",
423 rl["version"].get<std::string>(),
424 rl["status"].get<uint32_t>(),
425 rl["reason"].is_null() ? "" : rl["reason"].get<std::string>());
426
427 // Build the content to ensure we have the content-type
428 if (!rrd["content"].is_null() && rrd["content"].is_object()) {
429 body = rrd["content"].dump();
430 }
431 else if (!rrd["content"].is_null() && rrd["content"].is_string()) {
432 body = rrd["content"].get<std::string>();
433 }
434
435 // Headers..
436 for (auto& [k, v] : rrd["headers"].items()) {
437 if (v.is_string()) {
438 std::format_to(std::back_inserter(rs), "{}: {}\r\n", k, v.get<std::string>());
439 }
440 else if (v.is_number_unsigned()) {
441 std::format_to(std::back_inserter(rs), "{}: {}\r\n", k, v.get<uint64_t>());
442 }
443 else if (v.is_number_integer()) {
444 std::format_to(std::back_inserter(rs), "{}: {}\r\n", k, v.get<int>());
445 }
446 else {
447 std::format_to(std::back_inserter(rs), "{}: {}\r\n", k, v.dump());
448 }
449 }
450
451 std::format_to(std::back_inserter(rs), "\r\n");
452
453 // Finally the content..
454 if (!body.empty()) {
455 std::format_to(std::back_inserter(rs), "{}", body);
456 }
457
458 return std::move(rs);
459 }
