200 static std::string messageFromWininetCode(uint32_t errCode)
201 {
202 return WinInetErrorCodes.find(errCode) != WinInetErrorCodes.end()
203 ? std::format("{}-{}", errCode, WinInetErrorCodes[errCode])
204 : std::to_string(errCode);
205 }

245 void send(basic_restrequest&& req, std::function<void(const basic_restrequest&, const basic_restresponse&)>&& callback)
246 {
251 auto extractResponseLine = [](const std::wstring& src) -> std::tuple<std::string, uint32_t, std::string, size_t> {
252 // Given the first line is of the format: `HTTP_VERSION STATUS_CODE REASON_PHRASE\r\n`
253 // Return the particles.
254
255 // TODO: Optimize..
256 auto firstPart = src.find_first_of(L' ');
257 auto httpVersion = src.substr(0, firstPart);
258 auto secondPart = src.find_first_of(L' ', ++firstPart);
259 auto statusCode = src.substr(firstPart, secondPart - firstPart);
260 auto lastPart = src.find_first_of(L"\r\n", ++secondPart);
261 auto reasonPhrase = src.substr(secondPart, lastPart - secondPart);
262
263
264 return {ConversionUtils::asciiFromWide(httpVersion),
265 std::stoi(statusCode),
266 ConversionUtils::asciiFromWide(reasonPhrase),
267 lastPart + 2};
268 };
269
270 HRESULT hr {E_FAIL};
271 DWORD dwBytesRead {0}, dwError {0};
272 uint32_t nRetry {0}, nError {0};
273 char cBuf[MAXBUFSIZE] {};
274 DWORD dwFlagsSize = 0;
275 auto& hs = req["headers"]; // shortcut
276 auto& rs = req["request"]; // shortcut
277
278 // First order - adjust the UserAgent
279 if (!hs.contains("User-Agent")) req["headers"]["User-Agent"] = UserAgent;
280 auto strUserAgent = hs.contains("User-Agent") ? hs.value("User-Agent", UserAgent) : UserAgent;
281
282 if (hSession != NULL) {
283 auto& strServer = req.uri.authority.host;
284 if (ACW32HINTERNET hConnect {
285 WinHttpConnect(hSession, ConversionUtils::wideFromAscii(strServer).c_str(), req.uri.authority.port, 0)};
286 hConnect != NULL)
287 {
288 auto strMethod = rs.value("method", "");
289 auto strUrl = req.uri.urlPart;
290 auto strVersion = rs.value("version", "");
291
292 if (ACW32HINTERNET hRequest {WinHttpOpenRequest(hConnect,
293 ConversionUtils::wideFromAscii(strMethod).c_str(),
294 ConversionUtils::wideFromAscii(strUrl).c_str(),
295 ConversionUtils::wideFromAscii(strVersion).c_str(),
296 NULL,
297 RESTCL_ACCEPT_TYPES_W,
298 (req.uri.scheme == UriScheme::WebHttps)
299 ? WINHTTP_FLAG_SECURE | WINHTTP_FLAG_REFRESH
300 : WINHTTP_FLAG_REFRESH)};
301 hRequest != NULL)
302 {
303 auto contentLength = hs.value("Content-Length", 0);
304 std::string strHeaders;
305 req.encodeHeaders_to(strHeaders);
306 std::wstring requestHeaders = ConversionUtils::wideFromAscii(strHeaders);
307 nError = WinHttpAddRequestHeaders(hRequest,
308 requestHeaders.c_str(),
309 static_cast<DWORD>(requestHeaders.length()),
310 WINHTTP_ADDREQ_FLAG_ADD);
311
312 // Set this option so we don't incur a roundtrip delay
313 // ll = __LINE__;
314 // nError = WinHttpSetOption(
315 // hRequest, WINHTTP_OPTION_CLIENT_CERT_CONTEXT, WINHTTP_NO_CLIENT_CERT_CONTEXT, 0);
316
317 dwError = ERROR_SUCCESS;
318
319 nError = WinHttpSendRequest(hRequest,
320 WINHTTP_NO_ADDITIONAL_HEADERS,
321 0,
322 contentLength > 0 ? LPVOID(req.getContent().c_str()) : NULL,
323 contentLength,
324 contentLength,
325 NULL);
326
327 if (nError == FALSE, dwError = GetLastError(); dwError == ERROR_WINHTTP_CLIENT_AUTH_CERT_NEEDED) {
328 nError = WinHttpSetOption(
329 hRequest, WINHTTP_OPTION_CLIENT_CERT_CONTEXT, WINHTTP_NO_CLIENT_CERT_CONTEXT, 0);
330 if (nError == TRUE) {
331 nError = WinHttpSendRequest(hRequest,
332 WINHTTP_NO_ADDITIONAL_HEADERS,
333 0,
334 contentLength > 0 ? LPVOID(req.getContent().c_str()) : NULL,
335 contentLength,
336 contentLength,
337 NULL);
338 }
339 }
340
341
342 // Receive phase
343 RESTResponse resp;
344
345 // Get the "response" and the headers..
346 if (nError == FALSE) {
347 dwError = GetLastError();
348 }
349 else {
350 // Signals we should finish the request and wait for response from server.
351 nError = WinHttpReceiveResponse(hRequest, NULL);
352 if (nError == FALSE) {
353 dwError = GetLastError();
354 }
355 else {
356 // Get the headers next.. including the response line
357 DWORD dwSize = 0;
358
359 // First, use WinHttpQueryHeaders to obtain the size of the buffer.
360 WinHttpQueryHeaders(hRequest,
361 WINHTTP_QUERY_RAW_HEADERS_CRLF,
362 WINHTTP_HEADER_NAME_BY_INDEX,
363 NULL,
364 &dwSize,
365 WINHTTP_NO_HEADER_INDEX);
366
367 // Allocate memory for the buffer.
368 if (GetLastError() == ERROR_INSUFFICIENT_BUFFER) {
369 // A silly decision by the library to always return the headers as wchar_t
370 // despite the fact that over the wire we've got UTF-8/ASCII encoded headers
371 // The body is where encode varies and here in our library we focus on utf-8 json.
372 // The buffer must be wchar_t and we must adjust the size to account for the
373 // actual number of characters since the dwSize is in bytes not wchar_t characters!
374 auto lpOutBuffer = std::make_unique<wchar_t[]>(dwSize);
375
376 // Now, use WinHttpQueryHeaders to retrieve the header.
377 nError = WinHttpQueryHeaders(hRequest,
378 WINHTTP_QUERY_RAW_HEADERS_CRLF,
379 WINHTTP_HEADER_NAME_BY_INDEX,
380 lpOutBuffer.get(),
381 &dwSize,
382 WINHTTP_NO_HEADER_INDEX);
383 if (nError == TRUE && (lpOutBuffer != nullptr)) {
384 // The data is wchar_t but the dwSize returns bytes; adjustmend required.
385 dwSize /= sizeof(wchar_t);
386 // We need to skip the first line of the respose as it is the status response line.
387 // Decode the CRLF string into key-value elements.
388 std::wstring src {lpOutBuffer.get(), dwSize};
389
390 size_t startOfHeaders = std::string::npos;
391 std::tie(resp["response"]["version"],
392 resp["response"]["status"],
393 resp["response"]["reason"],
394 startOfHeaders) = extractResponseLine(src);
395 src.erase(0, startOfHeaders);
396 resp["headers"] =
397 string2map::parse<std::wstring, std::string, std::map<std::string, std::string>>(
398 src, L": ", L"\r\n");
399 }
400 }
401 }
402 }
403
404 // Next stage is to check for any errors and if none, get the body
405 if (dwError == ERROR_WINHTTP_NAME_NOT_RESOLVED) {
406 callback(req, RESTResponse {{dwError, messageFromWininetCode(dwError)}});
407 // throw(invalid_argument(
408 // std::format("HttpSendRequest() Failed Invalid host; dwError={}",
409 // messageFromWininetCode(dwError))));
410 }
411 else if ((dwError == ERROR_WINHTTP_CANNOT_CONNECT) || (dwError == ERROR_WINHTTP_CONNECTION_ERROR) ||
412 (dwError == ERROR_WINHTTP_OPERATION_CANCELLED) || (dwError == ERROR_WINHTTP_LOGIN_FAILURE) ||
413 (dwError == ERROR_WINHTTP_INVALID_SERVER_RESPONSE) || (dwError == ERROR_WINHTTP_RESEND_REQUEST) ||
414 (dwError == ERROR_WINHTTP_SECURE_FAILURE) || (dwError == ERROR_WINHTTP_TIMEOUT))
415 {
416 callback(req, RESTResponse {{dwError, messageFromWininetCode(dwError)}});
417 // throw io_exception(dwError,
418 // 0,
419 // std::format("HttpSendRequest() Failed to {}; dwError={}",
420 // argEndpoint,
421 // messageFromWininetCode(dwError)));
422 }
423 else if (dwError == ERROR_WINHTTP_INVALID_URL) {
424 callback(req, RESTResponse {{dwError, messageFromWininetCode(dwError)}});
425 // throw(invalid_argument(std::format("HttpSendRequest() Failed Invalid URL {}; dwError={}",
426 // argEndpoint,
427 // messageFromWininetCode(dwError))));
428 }
429 else if (dwError != ERROR_FILE_NOT_FOUND) {
430 nRetry = 0;
431 // while (dwError == ERROR_WINHTTP_CLIENT_AUTH_CERT_NEEDED)
432 //{
433 // nError = WinHttpSetOption(
434 // hRequest, WINHTTP_OPTION_CLIENT_CERT_CONTEXT, WINHTTP_NO_CLIENT_CERT_CONTEXT, 0);
435 //
436 // dwError = ERROR_SUCCESS;
437 // nError = WinHttpSendRequest(hRequest,
438 // !requestHeaders.empty() ? requestHeaders.c_str() : NULL,
439 // -1,
440 // argBody.has_value() ? (void*)argBody.value_or("").c_str() : NULL,
441 // DWORD(argBody.value_or("").length()),
442 // DWORD(argBody.value_or("").length()),
443 // NULL);
444 // dwError = GetLastError();
445 //
446 // nRetry++;
447 //
448 // if (nRetry > HTTPS_MAXRETRY) break;
449 //}
450
451 std::string rawResponse {};
452
453 do {
454 // memset(cBuf, '\0', sizeof(cBuf));
455 // Returns byte stream; accumulate until we're out of data.
456 hr = WinHttpReadData(hRequest, cBuf, MAXBUFSIZE - 1, &dwBytesRead)
457 ? S_OK
458 : HRESULT_FROM_WIN32(GetLastError());
459#pragma warning(suppress : 6102)
460 if (dwBytesRead) {
461 cBuf[dwBytesRead] = '\0';
462 rawResponse.append(cBuf, dwBytesRead);
463 }
464 } while (dwBytesRead > 0);
465
466 // Log everything; DELETE usually does not return a body!
467 hr = S_OK;
468 resp.setContent(rawResponse);
469
470 // Invoke the callback
471 callback(req, resp);
472 }
473 else {
474 callback(req, RESTResponse {{dwError, messageFromWininetCode(dwError)}});
475 // hr = HRESULT_FROM_WIN32(dwError);
476 // throw io_exception(dwError,
477 // 0,
478 // std::format("HttpSendRequest() Failed to {}; dwError={}",
479 // argEndpoint,
480 // messageFromWininetCode(dwError)));
481 }
482 }
483 else {
484 callback(req, RESTResponse {{hr, std::format("HttpOpenRequest() failed; dwError:{}", hr)}});
485 // throw io_exception(hr, 0, std::format("HttpOpenRequest() Failed; dwError={}", hr));
486 }
487 }
488 else {
489 callback(req, RESTResponse {{hr, std::format("WinHttpConnect() failed; dwError:{}", hr)}});
490 // throw io_exception(hr, 0, std::format("InternetConnect() Failed; dwError={}", hr));
491 }
492 }
493 else {
494 callback(req, RESTResponse {{hr, std::format("WinHttpOpen() failed; dwError:{}", hr)}});
495 // throw io_exception(hr, 0, std::format("InternetOpen() Failed; dwError={}", hr));
496 }
497 }
498 };
